name: Trading Bot Runner

on:
  schedule:
    # הקוד ירוץ כל יום ב-16:35 שעון ישראל (פתיחת המסחר בארה"ב)
    - cron: '35 14 * * *'
  workflow_dispatch: # מאפשר לך להריץ את הבוט ידנית בלחיצת כפתור

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Set up Python
        uses: actions/adjust-python-version@v4
        with:
          python-version: '3.9'

      - name: Install dependencies
        run: |
          pip install yfinance pandas requests

      - name: Run bot
        run: python bot.py
